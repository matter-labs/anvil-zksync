67pub fn bytes32_to_h256(data: Bytes32) -> H256 {
68    H256::from(data.as_u8_array_ref())

71pub fn h256_to_bytes32(data: &H256) -> Bytes32 {
72    Bytes32::from(*data.as_fixed_bytes())

75pub fn b160_to_h160(data: B160) -> H160 {
76    H160::from_slice(&data.to_be_bytes_vec())

79pub fn pad_to_word(input: &[u8]) -> Vec<u8> {
80    let mut data = input.to_owned();
81    let remainder = input.len().div_ceil(32) * 32 - input.len();
82    data.extend(std::iter::repeat_n(0u8, remainder));
83    data
84}

86pub fn h160_to_b160(data: &H160) -> B160 {
87    B160::from_be_bytes(*data.as_fixed_bytes())
88}

90// Helper methods to add data to the Vec<u8> in the format expected by ZKOS.
91pub fn append_address(data: &mut Vec<u8>, address: &H160) {
92    let mut pp = vec![0u8; 32];
93    let ap1 = address.as_fixed_bytes();
94    pp[12..(20 + 12)].copy_from_slice(ap1);
95    data.append(&mut pp);
96}
97
98pub fn append_u256(data: &mut Vec<u8>, payload: &zksync_types::U256) {
99    let mut pp = [0u8; 32];
100    payload.to_big_endian(&mut pp);
101
102    data.append(&mut pp.to_vec());
103}
104pub fn append_u64(data: &mut Vec<u8>, payload: u64) {
105    let mut pp = [0u8; 32];
106    let pp1 = payload.to_be_bytes();
107    pp[24..(8 + 24)].copy_from_slice(&pp1);
108    data.append(&mut pp.to_vec());
109}
110
111pub fn append_usize(data: &mut Vec<u8>, payload: usize) {
112    let mut pp = [0u8; 32];
113    let pp1 = payload.to_be_bytes();
114    pp[24..(8 + 24)].copy_from_slice(&pp1);
115    data.append(&mut pp.to_vec());
116}

118/// Iterates over raw storage and creates a tree from it.
119pub fn create_tree_from_full_state(
120    raw_storage: &InMemoryStorage,
121) -> (InMemoryTree, InMemoryPreimageSource) {
122    let original_state = &raw_storage.state;
123    let mut tree = InMemoryTree {
124        storage_tree: TestingTree::new_in(Global),
125        cold_storage: HashMap::new(),
126    };
127    let mut preimage_source = InMemoryPreimageSource {
128        inner: HashMap::new(),
129    };
130    for entry in &raw_storage.factory_deps {
131        preimage_source
132            .inner
133            .insert(h256_to_bytes32(entry.0), entry.1.clone());
134    }
135
136    for entry in original_state {
137        let kk = derive_flat_storage_key(
138            &h160_to_b160(entry.0.address()),
139            &h256_to_bytes32(entry.0.key()),
140        );
141        let vv = h256_to_bytes32(entry.1);
142
143        let aa = h256_to_bytes32(entry.0.key());
144        let cc = B160::try_from_be_slice(&aa.as_u8_array()[12..]).unwrap();
145
146        if entry.0.address() == &b160_to_h160(FAKE_NONCE_ADDRESS) {
147            set_account_properties(
148                &mut tree,
149                &mut preimage_source,
150                cc,
151                None,
152                Some(entry.1.to_low_u64_be()),
153                None,
154            );
155        }
156
157        if entry.0.address() == &b160_to_h160(BASE_TOKEN_ADDRESS) {
158            set_account_properties(
159                &mut tree,
160                &mut preimage_source,
161                cc,
162                Some(ruint::aliases::U256::from_be_slice(entry.1.as_bytes())),
163                None,
164                None,
165            );
166        }
167
168        if entry.0.address() == &b160_to_h160(ACCOUNT_CODE_STORAGE_STORAGE_ADDRESS) {
169            println!("Setting bytecode for {:?}", entry.0.key());
170            if entry.0.key().is_zero() {
171                continue;
172            }
173            if entry.1.is_zero() {
174                continue;
175            }
176
177            let bytecode_hash = entry.1;
178            let bytecode = raw_storage
179                .factory_deps
180                .get(bytecode_hash)
181                .unwrap_or_else(|| panic!("Cannot find bytecode for {:?}", bytecode_hash));
182            set_account_properties(
183                &mut tree,
184                &mut preimage_source,
185                cc,
186                None,
187                None,
188                Some(bytecode.clone()),
189            );
190        }
191
192        tree.storage_tree.insert(&kk, &vv);
193        tree.cold_storage.insert(kk, vv);
194    }
195
196    (tree, preimage_source)
197}

207// Serialize Transaction to ZKOS format.
208// Should match the code in basic_bootloader/src/bootloader/transaction/mod.rs
209pub fn transaction_to_zkos_vec(tx: &Transaction) -> Vec<u8> {
210    let mut tx_raw: Vec<u8> = vec![];
211    let tx_type_id = match tx.tx_format() {
212        zksync_types::l2::TransactionType::LegacyTransaction => 0u8,
213        zksync_types::l2::TransactionType::EIP2930Transaction => 1u8,
214        zksync_types::l2::TransactionType::EIP1559Transaction => 2u8,
215        zksync_types::l2::TransactionType::EIP712Transaction => todo!(),
216        zksync_types::l2::TransactionType::PriorityOpTransaction => todo!(),
217        zksync_types::l2::TransactionType::ProtocolUpgradeTransaction => todo!(),
218    };
219    let common_data = match &tx.common_data {
220        zksync_types::ExecuteTransactionCommon::L1(_) => todo!(),
221        zksync_types::ExecuteTransactionCommon::L2(l2_tx_common_data) => l2_tx_common_data,
222        zksync_types::ExecuteTransactionCommon::ProtocolUpgrade(_) => todo!(),
223    };
224    // tx_type
225    tx_raw.append(&mut vec![0u8; 31]);
226    tx_raw.append(&mut vec![tx_type_id; 1]);
227
228    // from
229    append_address(&mut tx_raw, &common_data.initiator_address);
230    // to
231    append_address(
232        &mut tx_raw,
233        &tx.execute.contract_address.unwrap_or(H160::zero()),
234    );
235
236    let gas_limit = common_data.fee.gas_limit;
237    // gas limit
238    append_u256(&mut tx_raw, &gas_limit);
239    // gas per pubdata limit
240    append_u256(&mut tx_raw, &common_data.fee.gas_per_pubdata_limit);
241
242    let fee_per_gas = common_data.fee.max_fee_per_gas;
243
244    // max fee per gas
245    append_u256(&mut tx_raw, &fee_per_gas);
246    // max priority fee per gas.
247    // TODO: hack for legacy tx (verify!!)
248    append_u256(&mut tx_raw, &common_data.fee.max_priority_fee_per_gas);
249
250    // paymaster
251    append_u64(&mut tx_raw, 0);
252
253    // nonce
254    append_u64(&mut tx_raw, common_data.nonce.0.into());
255
256    append_u256(&mut tx_raw, &tx.execute.value);
257
258    let mut reserved = [0u64; 4];
259
260    // TODO: should check chain_id
261    if tx_type_id == 0 {
262        reserved[0] = 1;
263    }
264
265    if tx.execute.contract_address.is_none() {
266        reserved[1] = 1;
267    }
268
269    for &value in reserved.iter().take(4) {
270        append_u64(&mut tx_raw, value);
271    }
272
273    let signature_u256 = common_data.signature.len().div_ceil(32) as u64;
274
275    let execute_calldata_words = tx.execute.calldata.len().div_ceil(32) as u64;
276
277    let mut current_offset = 19;
278
279    // data offset
280    append_u64(&mut tx_raw, current_offset * 32);
281    // lent
282    current_offset += 1 + execute_calldata_words;
283    // signature offset (stupid -- this doesn't include the padding!!)
284    append_u64(&mut tx_raw, current_offset * 32);
285    current_offset += 1 + signature_u256;
286
287    // factory deps
288    append_u64(&mut tx_raw, current_offset * 32);
289    current_offset += 1;
290    // paymaster
291    append_u64(&mut tx_raw, current_offset * 32);
292    current_offset += 1;
293    // reserved
294    append_u64(&mut tx_raw, current_offset * 32);
295
296    // len - data.
297    append_usize(&mut tx_raw, tx.execute.calldata.len());
298    tx_raw.append(&mut pad_to_word(&tx.execute.calldata));
299
300    // len - signature.
301    append_usize(&mut tx_raw, common_data.signature.len());
302    tx_raw.append(&mut pad_to_word(&common_data.signature));
303
304    // factory deps
305    append_u64(&mut tx_raw, 0);
306    // paymaster
307    append_u64(&mut tx_raw, 0);
308    // reserved
309    append_u64(&mut tx_raw, 0);
310    tx_raw
311}

313#[allow(clippy::too_many_arguments)]
314pub fn execute_tx_in_zkos<W: WriteStorage>(
315    tx: &Transaction,
316    tree: &InMemoryTree,
317    preimage_source: &InMemoryPreimageSource,
318    storage: &mut StoragePtr<W>,
319    simulate_only: bool,
320    batch_env: &L1BatchEnv,
321    chain_id: u64,
322    // if zkos_path is passed, it will also compute witness.
323    zkos_path: Option<String>,
324) -> (VmExecutionResultAndLogs, Option<Vec<u8>>) {
325    let batch_context = zk_ee::system::metadata::BlockMetadataFromOracle {
326        // TODO: get fee from batch_env.
327        eip1559_basefee: ruint::aliases::U256::from(if simulate_only { 0u64 } else { 1000u64 }),
328        block_number: batch_env.number.0 as u64,
329        timestamp: batch_env.timestamp,
330        gas_per_pubdata: ruint::aliases::U256::from(1u64),
331        chain_id,
332        // TODO: set proper values
333        block_hashes: Default::default(),
334        coinbase: B160::ZERO,
335        // TODO: investigate
336        native_price: ruint::aliases::U256::from(10u64),
337        gas_limit: 100_000_000,
338    };
339
340    let storage_commitment = StorageCommitment {
341        root: *tree.storage_tree.root(),
342        next_free_slot: tree.storage_tree.next_free_slot,
343    };
344
345    let mut tx = tx.clone();
346    if simulate_only {
347        // Currently zkos doesn't do validation when running a simulated transaction.
348        // This results in lower gas estimation - which might cause issues for the user.
349        // So here, we're manually adjusting the gas limit to account for the expected validation cost.
350        const ZKOS_EXPECTED_VALIDATION_COST: u64 = 6_000;
351        let new_gas_limit = tx
352            .gas_limit()
353            .saturating_sub(U256::from(ZKOS_EXPECTED_VALIDATION_COST));
354        match &mut tx.common_data {
355            ExecuteTransactionCommon::L1(data) => data.gas_limit = new_gas_limit,
356            ExecuteTransactionCommon::L2(data) => data.fee.gas_limit = new_gas_limit,
357            ExecuteTransactionCommon::ProtocolUpgrade(data) => data.gas_limit = new_gas_limit,
358        };
359
360        // FIXME: currently zkos requires gas price to be >0 (otherwise it fails with out of resources during validation)
361        if tx.max_fee_per_gas() == 0.into() {
362            let new_gas_price = 100.into();
363            if let ExecuteTransactionCommon::L2(data) = &mut tx.common_data {
364                data.fee.max_fee_per_gas = new_gas_price;
365                data.fee.max_priority_fee_per_gas = new_gas_price;
366            };
367        }
368    }
369
370    let tx_raw = transaction_to_zkos_vec(&tx);
371
372    let mut witness = None;
373
374    let (output, dynamic_factory_deps, storage_logs) = if simulate_only {
375        (
376            forward_system::run::simulate_tx(
377                tx_raw,
378                batch_context,
379                tree.clone(),
380                preimage_source.clone(),
381            )
382            .unwrap(),
383            Default::default(), // dynamic factory deps
384            vec![],             // storage logs
385        )
386    } else {
387        let tx_source = TxListSource {
388            transactions: vec![tx_raw].into(),
389        };
390        let noop = NoopTxCallback {};
391        let batch_output = forward_system::run::run_batch(
392            batch_context,
393            // TODO: FIXME
394            tree.clone(),
395            preimage_source.clone(),
396            tx_source.clone(),
397            noop,
398        )
399        .unwrap();
400
401        if let Some(zkos_path) = zkos_path {
402            let result = zksync_os_api::run_batch_generate_witness(
403                batch_context,
404                tree.clone(),
405                preimage_source.clone(),
406                tx_source,
407                storage_commitment,
408                &zkos_path,
409            );
410
411            witness = Some(result.iter().flat_map(|x| x.to_be_bytes()).collect());
412        }
413
414        let mut storage_ptr = storage.borrow_mut();
415
416        let mut storage_logs = vec![];
417
418        // apply storage writes..
419        for write in batch_output.storage_writes {
420            let storage_key = StorageKey::new(
421                AccountTreeId::new(Address::from_slice(&write.account.to_be_bytes_vec())),
422                H256::from(write.account_key.as_u8_array_ref()),
423            );
424            let storage_value = H256::from(write.value.as_u8_array_ref());
425            let prev_value = storage_ptr.set_value(storage_key, storage_value);
426
427            let storage_log = StorageLog {
428                // TODO: FIXME - should distinguish between initial write and repeated write.
429                kind: zksync_types::StorageLogKind::InitialWrite,
430                key: storage_key,
431                value: storage_value,
432            };
433            storage_logs.push(StorageLogWithPreviousValue {
434                log: storage_log,
435                previous_value: prev_value,
436            });
437            if write.account == ACCOUNT_PROPERTIES_STORAGE_ADDRESS {
438                // also update a balance & nonce.
439
440                // find preimage in batch output.published_preimages
441                let dst_address = &write.account_key.as_u8_ref()[12..];
442                let dst_address = zksync_types::Address::from_slice(dst_address);
443                let mut found = false;
444
445                for preimage in batch_output.published_preimages.iter() {
446                    if preimage.0 == write.value {
447                        found = true;
448                        let account_properties =
449                            AccountProperties::decode(&preimage.1.clone().try_into().unwrap());
450                        let balance = account_properties.balance;
451                        let nonce = account_properties.nonce;
452                        let bytecode_hash = account_properties.bytecode_hash;
453
454                        println!(
455                            "For account :{:?} nonce is {:?} balance is {:?}, bytecode_hash is {:?} preimage hash is {:?}",
456                            dst_address, nonce, balance, bytecode_hash, preimage.0
457                        );
458
459                        for (key, value) in [
460                            (
461                                zksync_os_get_nonce_key(&dst_address),
462                                u256_to_h256(U256::from(nonce)),
463                            ),
464                            (
465                                zkos_storage_key_for_standard_token_balance(
466                                    AccountTreeId::new(b160_to_h160(BASE_TOKEN_ADDRESS)),
467                                    &dst_address,
468                                ),
469                                H256::from_slice(&balance.to_be_bytes_vec()),
470                            ),
471                            (
472                                get_code_key(&dst_address),
473                                H256::from(bytecode_hash.as_u8_array_ref()),
474                            ),
475                        ] {
476                            let storage_log = StorageLog {
477                                // TODO: FIXME - should distinguish between initial write and repeated write.
478                                kind: zksync_types::StorageLogKind::InitialWrite,
479                                key,
480                                value,
481                            };
482                            // Create storage write, only if actually something has changed.
483                            let prev_value = storage_ptr.read_value(&key);
484                            if prev_value != value {
485                                let prev_value = storage_ptr.set_value(key, value);
486
487                                storage_logs.push(StorageLogWithPreviousValue {
488                                    log: storage_log,
489                                    previous_value: prev_value,
490                                })
491                            };
492                        }
493                    }
494                }
495                if !found {
496                    panic!("Failed to find preimage for account_properties");
497                }
498            }
499        }
500        let mut f_deps = HashMap::new();
501
502        for factory_dep in batch_output.published_preimages {
503            f_deps.insert(bytes32_to_h256(factory_dep.0), factory_dep.1);
504        }
505
506        (batch_output.tx_results[0].clone(), f_deps, storage_logs)
507    };
508
509    let (tx_output, gas_refunded) = match output.as_ref() {
510        Ok(tx_output) => match &tx_output.execution_result {
511            forward_system::run::ExecutionResult::Success(output) => match &output {
512                forward_system::run::ExecutionOutput::Call(data) => (data, tx_output.gas_refunded),
513                forward_system::run::ExecutionOutput::Create(data, _) => {
514                    (data, tx_output.gas_refunded)
515                }
516            },
517            forward_system::run::ExecutionResult::Revert(data) => {
518                return (
519                    VmExecutionResultAndLogs {
520                        result: ExecutionResult::Revert {
521                            output: VmRevertReason::General {
522                                msg: "Transaction reverted".to_string(),
523                                data: data.clone(),
524                            },
525                        },
526                        logs: VmExecutionLogs {
527                            // TODO: check if we should return storage_logs on revert.
528                            storage_logs,
529                            events: Default::default(),
530                            user_l2_to_l1_logs: Default::default(),
531                            system_l2_to_l1_logs: Default::default(),
532                            total_log_queries_count: Default::default(),
533                        },
534                        statistics: Default::default(),
535                        refunds: Refunds {
536                            gas_refunded: tx_output.gas_refunded,
537                            operator_suggested_refund: tx_output.gas_refunded,
538                        },
539                        dynamic_factory_deps,
540                    },
541                    witness,
542                );
543            }
544        },
545        Err(invalid_tx) => {
546            return (
547                VmExecutionResultAndLogs {
548                    result: ExecutionResult::Revert {
549                        output: VmRevertReason::General {
550                            msg: format!("{:?}", invalid_tx),
551                            data: vec![],
552                        },
553                    },
554                    logs: Default::default(),
555                    statistics: Default::default(),
556                    refunds: Default::default(),
557                    dynamic_factory_deps: Default::default(),
558                },
559                witness,
560            )
561        }
562    };
563
564    (
565        VmExecutionResultAndLogs {
566            result: ExecutionResult::Success {
567                output: tx_output.clone(),
568            },
569            logs: VmExecutionLogs {
570                storage_logs,
571                events: Default::default(),
572                user_l2_to_l1_logs: Default::default(),
573                system_l2_to_l1_logs: Default::default(),
574                total_log_queries_count: Default::default(),
575            },
576            statistics: Default::default(),
577            refunds: Refunds {
578                gas_refunded,
579                operator_suggested_refund: gas_refunded,
580            },
581            dynamic_factory_deps,
582        },
583        witness,
584    )
585}

587pub fn zksync_os_get_nonce_key(account: &Address) -> StorageKey {
588    let nonce_manager = AccountTreeId::new(b160_to_h160(FAKE_NONCE_ADDRESS));
589
590    // The `minNonce` (used as nonce for EOAs) is stored in a mapping inside the `NONCE_HOLDER` system contract
591    let key = address_to_h256(account);
592
593    StorageKey::new(nonce_manager, key)
594}

600/// Create a `key` part of `StorageKey` to access the balance from ERC20 contract balances
601fn zksync_os_key_for_erc20_balance(address: &Address) -> H256 {
602    let address_h256 = address_to_h256(address);
603
604    // 20 bytes address first gets aligned to 32 bytes with index of `balanceOf` storage slot
605    // of default ERC20 contract and to then to 64 bytes.
606    let slot_index = H256::from_low_u64_be(51);
607    let mut bytes = [0_u8; 64];
608    bytes[..32].copy_from_slice(address_h256.as_bytes());
609    bytes[32..].copy_from_slice(slot_index.as_bytes());
610    H256(keccak256(&bytes))
611}

613fn zkos_storage_key_for_standard_token_balance(
614    token_contract: AccountTreeId,
615    address: &Address,
616) -> StorageKey {
617    // We have different implementation of the standard ERC20 contract and native
618    // eth contract. The key for the balance is different for each.
619    let key = if token_contract.address() == &b160_to_h160(BASE_TOKEN_ADDRESS) {
620        zksync_os_key_for_eth_balance(address)
621    } else {
622        zksync_os_key_for_erc20_balance(address)
623    };
624
625    StorageKey::new(token_contract, key)

828fn get_account_properties(
829    state_tree: &InMemoryTree,
830    preimage_source: &mut InMemoryPreimageSource,
831    address: &B160,
832) -> AccountProperties {
833    use forward_system::run::PreimageSource;
834    let key = address_into_special_storage_key(address);
835    let flat_key = derive_flat_storage_key(&ACCOUNT_PROPERTIES_STORAGE_ADDRESS, &key);
836    match state_tree.cold_storage.get(&flat_key) {
837        None => AccountProperties::default(),
838        Some(account_hash) => {
839            if account_hash.is_zero() {
840                // Empty (default) account
841                AccountProperties::default()
842            } else {
843                // Get from preimage:
844                let encoded = preimage_source
845                    .get_preimage(*account_hash)
846                    .unwrap_or_else(|| {
847                        panic!("Missing preimage for {:?} and {:?}", address, account_hash)
848                    });
849
850                AccountProperties::decode(&encoded.try_into().unwrap())
851            }
852        }
853    }
854}
855
856pub fn set_account_properties(
857    state_tree: &mut InMemoryTree,
858    preimage_source: &mut InMemoryPreimageSource,
859
860    address: B160,
861    balance: Option<ruint::aliases::U256>,
862    nonce: Option<u64>,
863    bytecode: Option<Vec<u8>>,
864) {
865    let mut account_properties = get_account_properties(state_tree, preimage_source, &address);
866    if let Some(bytecode) = bytecode {
867        account_properties = evm_bytecode_into_account_properties(&bytecode);
868        // Save bytecode preimage
869        preimage_source
870            .inner
871            .insert(account_properties.bytecode_hash, bytecode);
872    }
873    if let Some(nominal_token_balance) = balance {
874        account_properties.balance = nominal_token_balance;
875    }
876    if let Some(nonce) = nonce {
877        account_properties.nonce = nonce;
878    }
879
880    let encoding = account_properties.encoding();
881    let properties_hash = account_properties.compute_hash();
882
883    let key = address_into_special_storage_key(&address);
884    let flat_key = derive_flat_storage_key(&ACCOUNT_PROPERTIES_STORAGE_ADDRESS, &key);
885
886    // Save preimage
887    preimage_source
888        .inner
889        .insert(properties_hash, encoding.to_vec());
890    state_tree.cold_storage.insert(flat_key, properties_hash);
891    state_tree.storage_tree.insert(&flat_key, &properties_hash);
892}

894/// Copied from https://github.com/matter-labs/zksync-os/blob/ab9f5d1856450d688f59996c04830e25c1824e95/tests/rig/src/utils.rs#L425
895/// To avoid bringing in `rig` as a dependency.
896pub fn evm_bytecode_into_account_properties(bytecode: &[u8]) -> AccountProperties {
897    use crypto::blake2s::Blake2s256;
898    use crypto::sha3::Keccak256;
899    use crypto::MiniDigest;
900
901    let observable_bytecode_hash = Bytes32::from_array(Keccak256::digest(bytecode));
902    let bytecode_hash = Bytes32::from_array(Blake2s256::digest(bytecode));
903    let mut result = AccountProperties::TRIVIAL_VALUE;
904    result.observable_bytecode_hash = observable_bytecode_hash;
905    result.bytecode_hash = bytecode_hash;
906    result.versioning_data.set_as_deployed();
907    result
908        .versioning_data
909        .set_ee_version(ExecutionEnvironmentType::EVM as u8);
910    result
911        .versioning_data
912        .set_code_version(DEFAULT_CODE_VERSION_BYTE);
913    result.bytecode_len = bytecode.len() as u32;
914    result.artifacts_len = 0;
915    result.observable_bytecode_len = bytecode.len() as u32;
916
917    result
