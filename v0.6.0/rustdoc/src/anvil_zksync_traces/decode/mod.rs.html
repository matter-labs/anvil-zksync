31/// The first four bytes of the call data for a function call specifies the function to be called.
32pub const SELECTOR_LEN: usize = 4;

34/// A base struct for builders. If you want to start building from a predefined
35/// [CallTraceDecoder] instance, create your own builder like that:
36///
37/// ```rust,ignore
38/// pub struct CallTraceDecoderBuilder;
39/// impl CallTraceDecoderBuilder {
40/// #[inline]
41/// pub fn default() -> CallTraceDecoderBuilderBase {
42/// // Construct an instance here
43/// }
44/// }
45/// ```
46#[must_use = "builders do nothing unless you call `build` on them"]
47pub struct CallTraceDecoderBuilderBase {
48    decoder: CallTraceDecoder,
49}
50
51impl CallTraceDecoderBuilderBase {
52    /// Create a new builder.
53    #[inline]
54    pub fn new(starting_decoder_state: CallTraceDecoder) -> Self {
55        Self {
56            decoder: starting_decoder_state,
57        }
58    }
59
60    /// Add known labels to the decoder.
61    #[inline]
62    pub fn with_labels(mut self, labels: impl IntoIterator<Item = (Address, String)>) -> Self {
63        self.decoder.labels.extend(labels);
64        self
65    }
66
67    /// Sets the signature identifier for events and functions.
68    #[inline]
69    pub fn with_signature_identifier(mut self, identifier: SingleSignaturesIdentifier) -> Self {
70        self.decoder.signature_identifier = Some(identifier);
71        self
72    }
73
74    /// Build the decoder.
75    #[inline]
76    pub fn build(self) -> CallTraceDecoder {
77        self.decoder
78    }
79}

81/// The call trace decoder.
82///
83/// The decoder collects address labels which it
84/// then uses to decode the call trace.
85#[derive(Clone, Debug, Default)]
86pub struct CallTraceDecoder {
87    /// Addresses identified to be a specific contract.
88    ///
89    /// The values are in the form `"<artifact>:<contract>"`.
90    pub contracts: HashMap<Address, String>,
91    /// Address labels.
92    pub labels: HashMap<Address, String>,
93    /// Contract addresses that have a receive function.
94    pub receive_contracts: Vec<Address>,
95    /// Contract addresses that have fallback functions, mapped to function sigs.
96    pub fallback_contracts: HashMap<Address, Vec<String>>,
97    /// All known events.
98    pub events: BTreeMap<(B256, usize), Vec<Event>>,
99    /// Revert decoder. Contains all known custom errors.
100    pub revert_decoder: RevertDecoder,
101    /// All known functions.
102    pub functions: HashMap<Selector, Vec<Function>>,
103    /// A signature identifier for events and functions.
104    pub signature_identifier: Option<SingleSignaturesIdentifier>,
105}

107impl CallTraceDecoder {
108    /// Creates a new call trace decoder.
109    pub fn new(
110        functions: HashMap<Selector, Vec<Function>>,
111        events: BTreeMap<(B256, usize), Vec<Event>>,
112    ) -> Self {
113        // Add known addresses (system contracts, precompiles) to the labels
114        let labels: HashMap<H160, String> = KNOWN_ADDRESSES
115            .iter()
116            .map(|(address, known_address)| (*address, known_address.name.clone()))
117            .collect();
118
119        Self {
120            contracts: Default::default(),
121            labels,
122            receive_contracts: Default::default(),
123            fallback_contracts: Default::default(),
124            functions,
125            events,
126            revert_decoder: Default::default(),
127            signature_identifier: None,
128        }
129    }

131    /// Populates the traces with decoded data by mutating the
132    /// [CallTrace] in place. See [CallTraceDecoder::decode_function] and
133    /// [CallTraceDecoder::decode_event] for more details.
134    pub async fn populate_traces(
135        &self,
136        traces: &mut Vec<CallTraceNode>,
137    ) -> Result<(), DecodingError> {
138        for node in traces {
139            node.trace.decoded = self.decode_function(&node.trace).await?;
140            for log in node.logs.iter_mut() {
141                log.decoded = self.decode_event(&log.raw_log).await;
142            }
143        }
144        Ok(())
145    }

147    /// Decodes a call trace.
148    pub async fn decode_function(
149        &self,
150        trace: &CallTrace,
151    ) -> Result<DecodedCallTrace, DecodingError> {
152        let label = self.labels.get(&trace.address).cloned();
153        let cdata = &trace.call.input;
154
155        if cdata.len() >= SELECTOR_LEN {
156            let selector = &cdata[..SELECTOR_LEN];
157            let mut functions = Vec::new();
158            let functions = match self.functions.get(selector) {
159                Some(fs) => fs,
160                None => {
161                    if let Some(identifier) = &self.signature_identifier {
162                        if let Some(function) =
163                            identifier.write().await.identify_function(selector).await
164                        {
165                            functions.push(function);
166                        }
167                    }
168                    &functions
169                }
170            };
171            let [func, ..] = &functions[..] else {
172                return Ok(DecodedCallTrace {
173                    label,
174                    call_data: None,
175                    return_data: self.default_return_data(trace),
176                });
177            };
178
179            // If traced contract is a fallback contract, check if it has the decoded function.
180            // If not, then replace call data signature with `fallback`.
181            let mut call_data = self.decode_function_input(trace, func);
182            if let Some(fallback_functions) = self.fallback_contracts.get(&trace.address) {
183                if !fallback_functions.contains(&func.signature()) {
184                    call_data.signature = "fallback()".into();
185                }
186            }
187
188            Ok(DecodedCallTrace {
189                label,
190                call_data: Some(call_data),
191                return_data: self.decode_function_output(trace, functions),
192            })
193        } else {
194            let has_receive = self.receive_contracts.contains(&trace.address);
195            let signature = if cdata.is_empty() && has_receive {
196                "receive()"
197            } else {
198                "fallback()"
199            }
200            .into();
201            let args = if cdata.is_empty() {
202                Vec::new()
203            } else {
204                vec![DecodedValue::Bytes(cdata.clone())]
205            };
206            Ok(DecodedCallTrace {
207                label,
208                call_data: Some(DecodedCallData { signature, args }),
209                return_data: self.default_return_data(trace),
210            })
211        }
212    }

214    /// Decodes a function's input into the given trace.
215    fn decode_function_input(&self, trace: &CallTrace, func: &Function) -> DecodedCallData {
216        let mut args = None;
217        if trace.call.input.len() >= SELECTOR_LEN && args.is_none() {
218            if let Ok(v) = func.abi_decode_input(&trace.call.input[SELECTOR_LEN..], false) {
219                args = Some(
220                    v.into_iter()
221                        .flat_map(|value| -> Result<DecodedValue, DecodingError> {
222                            let decoded = decode_value(value)?;
223                            Ok(label_value(decoded, |value| {
224                                self.labels.get(value).cloned()
225                            }))
226                        })
227                        .collect(),
228                );
229            }
230        }
231        DecodedCallData {
232            signature: func.signature(),
233            args: args.unwrap_or_default(),
234        }
235    }

237    /// Decodes a function's output into the given trace.
238    fn decode_function_output(&self, trace: &CallTrace, funcs: &[Function]) -> DecodedReturnData {
239        if !trace.success {
240            return self.default_return_data(trace);
241        }
242
243        if let Some(values) = funcs
244            .iter()
245            .find_map(|func| func.abi_decode_output(&trace.call.output, false).ok())
246        {
247            // Functions coming from an external database do not have any outputs specified,
248            // and will lead to returning an empty list of values.
249            if values.is_empty() {
250                return DecodedReturnData::NormalReturn(vec![]);
251            }
252
253            return DecodedReturnData::NormalReturn(
254                values
255                    .into_iter()
256                    .flat_map(|value| self.decode_value(value))
257                    .collect(),
258            );
259        }
260        DecodedReturnData::NormalReturn(vec![])
261    }

263    /// Decodes an event from ZKsync type VmEvent.
264    pub async fn decode_event(&self, vm_event: &VmEvent) -> DecodedCallEvent {
265        let Some(&t0) = vm_event.indexed_topics.first() else {
266            return DecodedCallEvent {
267                name: None,
268                params: None,
269            };
270        };
271
272        let mut events = Vec::new();
273        let b256_t0 = B256::from_slice(t0.as_bytes());
274        let key = (b256_t0, indexed_inputs_zksync(vm_event) - 1);
275        let events = match self.events.get(&key) {
276            Some(es) => es,
277            None => {
278                if let Some(identifier) = &self.signature_identifier {
279                    if let Some(event) = identifier.write().await.identify_event(&t0[..]).await {
280                        events.push(get_indexed_event_from_vm_event(event, vm_event));
281                    }
282                }
283                &events
284            }
285        };
286        let log_data = vm_event_to_log_data(vm_event);
287        for event in events {
288            if let Ok(decoded) = event.decode_log(&log_data, false) {
289                let params = reconstruct_params(event, &decoded);
290                return DecodedCallEvent {
291                    name: Some(event.name.clone()),
292                    params: Some(
293                        params
294                            .into_iter()
295                            .zip(event.inputs.iter())
296                            .flat_map(
297                                |(param, input)| -> Result<(String, DecodedValue), DecodingError> {
298                                    // undo patched names
299                                    let name: String = input.name.clone();
300                                    let value: DecodedValue = self.decode_value(param)?;
301                                    Ok((name, value))
302                                },
303                            )
304                            .collect(),
305                    ),
306                };
307            }
308        }
309
310        DecodedCallEvent {
311            name: None,
312            params: None,
313        }
314    }

316    /// Prefetches function and event signatures into the identifier cache
317    pub async fn prefetch_signatures(&self, nodes: &[CallTraceNode]) {
318        let Some(identifier) = &self.signature_identifier else {
319            return;
320        };
321
322        let events: Vec<_> = nodes
323            .iter()
324            .flat_map(|node| {
325                node.logs
326                    .iter()
327                    .filter_map(|log| log.raw_log.indexed_topics.first().cloned())
328            })
329            .unique()
330            .collect();
331        identifier.write().await.identify_events(events).await;
332
333        let funcs: Vec<_> = nodes
334            .iter()
335            .filter_map(|n| n.trace.call.input.get(..SELECTOR_LEN).map(|s| s.to_vec()))
336            .filter(|s| !self.functions.contains_key(s.as_slice()))
337            .collect();
338        identifier.write().await.identify_functions(funcs).await;
339
340        // Need to decode revert reasons and errors as well
341    }

343    /// The default decoded return data for a trace.
344    fn default_return_data(&self, trace: &CallTrace) -> DecodedReturnData {
345        if trace.success {
346            DecodedReturnData::NormalReturn(vec![])
347        } else {
348            DecodedReturnData::Revert(DecodedRevertData::Error(
349                self.revert_decoder.decode(&trace.call.output),
350            ))
351        }
352    }

354    fn decode_value(&self, value: DynSolValue) -> Result<DecodedValue, DecodingError> {
355        decode_value(value).map(|value| label_value(value, |addr| self.labels.get(addr).cloned()))
356    }

359pub fn label_value(
360    value: DecodedValue,
361    labeler: impl Fn(&Address) -> Option<String>,
362) -> DecodedValue {
363    fn label_values(
364        vec: Vec<DecodedValue>,
365        labeler: &dyn Fn(&Address) -> Option<String>,
366    ) -> Vec<DecodedValue> {
367        vec.into_iter().map(|v| label_value(v, labeler)).collect()
368    }
369
370    match value {
371        DecodedValue::Address(LabeledAddress {
372            label: None,
373            address,
374        }) => {
375            let label = labeler(&address);
376            if let Some(label) = &label {
377                tracing::info!("Address {address:?} resolved to the label {label}.");
378            };
379            DecodedValue::Address(LabeledAddress { label, address })
380        }
381        DecodedValue::Array(vec) => DecodedValue::Array(label_values(vec, &labeler)),
382        DecodedValue::FixedArray(vec) => DecodedValue::FixedArray(label_values(vec, &labeler)),
383        DecodedValue::Tuple(vec) => DecodedValue::Tuple(label_values(vec, &labeler)),
384        DecodedValue::CustomStruct {
385            name,
386            prop_names,
387            tuple,
388        } => DecodedValue::CustomStruct {
389            name,
390            prop_names,
391            tuple: label_values(tuple, &labeler),
392        },
393        other => other,
394    }
395}

397pub fn decode_value(value: DynSolValue) -> Result<DecodedValue, DecodingError> {
398    match value {
399        DynSolValue::Bool(b) => Ok(DecodedValue::Bool(b)),
400        DynSolValue::Int(i, _) => Ok(match i.into_sign_and_abs() {
401            (Sign::Positive, value) => {
402                DecodedValue::Int(zksync_types::U256(value.into_limbs()).into())
403            }
404            (Sign::Negative, value) => DecodedValue::Int(SignedU256 {
405                sign: anvil_zksync_types::numbers::Sign::Negative,
406                inner: zksync_types::U256(value.into_limbs()),
407            }),
408        }),
409        DynSolValue::Uint(u, _) => Ok(DecodedValue::Uint(zksync_types::U256(u.into_limbs()))),
410        DynSolValue::FixedBytes(word, size) => {
411            // Convert Word to Word32 (assuming proper conversion exists)
412            let word32 = Word32::from(word); // This assumes there's a conversion method
413            Ok(DecodedValue::FixedBytes(word32, size))
414        }
415        DynSolValue::Address(addr) => match <[u8; 20]>::try_from(addr.0.as_slice()) {
416            Ok(raw_bytes_20) => {
417                let address = Address::from(raw_bytes_20);
418                Ok(DecodedValue::Address(LabeledAddress {
419                    label: None,
420                    address,
421                }))
422            }
423            Err(_e) => Err(DecodingError::InvalidAddress {
424                address: format!("{addr:?}"),
425            }),
426        },
427        DynSolValue::Function(func) => Ok(DecodedValue::Function(*func.0)),
428        DynSolValue::Bytes(bytes) => Ok(DecodedValue::Bytes(bytes)),
429        DynSolValue::String(s) => Ok(DecodedValue::String(s)),
430        DynSolValue::Array(arr) => {
431            let decoded = arr.into_iter().flat_map(decode_value).collect();
432            Ok(DecodedValue::Array(decoded))
433        }
434        DynSolValue::FixedArray(arr) => {
435            let decoded = arr.into_iter().flat_map(decode_value).collect();
436            Ok(DecodedValue::FixedArray(decoded))
437        }
438        DynSolValue::Tuple(tup) => {
439            let decoded = tup.into_iter().flat_map(decode_value).collect();
440            Ok(DecodedValue::Tuple(decoded))
441        }
442        DynSolValue::CustomStruct {
443            name,
444            prop_names,
445            tuple,
446        } => {
447            let decoded = tuple.into_iter().flat_map(decode_value).collect();
448            Ok(DecodedValue::CustomStruct {
449                name,
450                prop_names,
451                tuple: decoded,
452            })
453        }
454    }
455}

457/// Restore the order of the params of a decoded event,
458/// as Alloy returns the indexed and unindexed params separately.
459fn reconstruct_params(event: &Event, decoded: &DecodedEvent) -> Vec<DynSolValue> {
460    let mut indexed = 0;
461    let mut unindexed = 0;
462    let mut inputs = vec![];
463    for input in event.inputs.iter() {
464        // Prevent panic of event `Transfer(from, to)` decoded with a signature
465        // `Transfer(address indexed from, address indexed to, uint256 indexed tokenId)` by making
466        // sure the event inputs is not higher than decoded indexed / un-indexed values.
467        if input.indexed && indexed < decoded.indexed.len() {
468            inputs.push(decoded.indexed[indexed].clone());
469            indexed += 1;
470        } else if unindexed < decoded.body.len() {
471            inputs.push(decoded.body[unindexed].clone());
472            unindexed += 1;
473        }
474    }
475
476    inputs
477}
478fn indexed_inputs_zksync(event: &VmEvent) -> usize {
479    event.indexed_topics.len()
480}

482/// Given an `Event` without indexed parameters and a `VmEvent`, it tries to
483/// return the `Event` with the proper indexed parameters. Otherwise,
484/// it returns the original `Event`.
485pub fn get_indexed_event_from_vm_event(mut event: Event, vm_event: &VmEvent) -> Event {
486    if !event.anonymous && vm_event.indexed_topics.len() > 1 {
487        let indexed_params = vm_event.indexed_topics.len() - 1;
488        let num_inputs = event.inputs.len();
489        let num_address_params = event.inputs.iter().filter(|p| p.ty == "address").count();
490
491        event
492            .inputs
493            .iter_mut()
494            .enumerate()
495            .for_each(|(index, param)| {
496                if param.name.is_empty() {
497                    param.name = format!("param{index}");
498                }
499                if num_inputs == indexed_params
500                    || (num_address_params == indexed_params && param.ty == "address")
501                {
502                    param.indexed = true;
503                }
504            })
505    }
506    event
507}

509/// Converts a `VmEvent` to a `LogData`.
510pub fn vm_event_to_log_data(event: &VmEvent) -> LogData {
511    LogData::new_unchecked(
512        event
513            .indexed_topics
514            .iter()
515            .map(|h| B256::from_slice(h.as_bytes()))
516            .collect(),
517        event.value.clone().into(),
518    )
