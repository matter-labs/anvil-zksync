149    /// Create [L1BatchEnv] to be used in the VM.
150    ///
151    /// We compute l1/l2 block details from storage to support fork testing, where the storage
152    /// can be updated mid execution and no longer matches with the initial node's state.
153    /// The L1 & L2 timestamps are also compared with node's timestamp to ensure it always increases monotonically.
154    pub async fn create_l1_batch_env(&self) -> (L1BatchEnv, BlockContext) {
155        tracing::debug!("creating L1 batch env");
156
157        let (last_l1_batch_number, last_l2_block) = self.blockchain.read().await.last_env(
158            &StorageView::new(&self.fork_storage).to_rc_ptr(),
159            &self.time,
160        );
161
162        let block_ctx = BlockContext {
163            hash: H256::zero(),
164            batch: (last_l1_batch_number + 1).0,
165            miniblock: last_l2_block.number as u64 + 1,
166            timestamp: self.time.peek_next_timestamp(),
167            prev_block_hash: last_l2_block.hash,
168        };
169
170        let fee_input = if let Some(fork_details) = self.fork.details() {
171            // TODO: This is a weird pattern. `TestNodeFeeInputProvider` should encapsulate fork's
172            //       behavior by taking fork's fee input into account during initialization.
173            BatchFeeInput::PubdataIndependent(PubdataIndependentBatchFeeModelInput {
174                l1_gas_price: fork_details.l1_gas_price,
175                fair_l2_gas_price: fork_details.l2_fair_gas_price,
176                fair_pubdata_price: fork_details.fair_pubdata_price,
177            })
178        } else {
179            self.fee_input_provider.get_batch_fee_input()
180        };
181
182        let batch_env = L1BatchEnv {
183            // TODO: set the previous batch hash properly (take from fork, when forking, and from local storage, when this is not the first block).
184            previous_batch_hash: None,
185            number: L1BatchNumber::from(block_ctx.batch),
186            timestamp: block_ctx.timestamp,
187            fee_input,
188            fee_account: H160::zero(),
189            enforced_base_fee: None,
190            first_l2_block: L2BlockEnv {
191                // the 'current_miniblock' contains the block that was already produced.
192                // So the next one should be one higher.
193                number: block_ctx.miniblock as u32,
194                timestamp: block_ctx.timestamp,
195                prev_block_hash: last_l2_block.hash,
196                // This is only used during zksyncEra block timestamp/number transition.
197                // In case of starting a new network, it doesn't matter.
198                // In theory , when forking mainnet, we should match this value
199                // to the value that was set in the node at that time - but AFAIK
200                // we don't have any API for this - so this might result in slightly
201                // incorrect replays of transacions during the migration period, that
202                // depend on block number or timestamp.
203                max_virtual_blocks_to_create: 1,
204            },
205        };
206
207        (batch_env, block_ctx)
208    }

210    #[allow(clippy::too_many_arguments)]
211    async fn apply_batch(
212        &mut self,
213        batch_timestamp: u64,
214        base_system_contracts_hashes: BaseSystemContractsHashes,
215        block: api::Block<api::TransactionVariant>,
216        virtual_block: Option<api::Block<api::TransactionVariant>>,
217        tx_results: Vec<TransactionResult>,
218        finished_l1_batch: FinishedL1Batch,
219        modified_storage_keys: HashMap<StorageKey, StorageValue>,
220    ) {
221        // TODO: `apply_batch` is leaking a lot of abstractions and should be wholly contained inside `Blockchain`.
222        //       Additionally, a dedicated `PreviousStates` struct would help with separation of concern.
223        /// Archives the current state for later queries.
224        fn archive_state(
225            previous_states: &mut IndexMap<H256, HashMap<StorageKey, StorageValue>>,
226            state: HashMap<StorageKey, StorageValue>,
227            block_number: L2BlockNumber,
228            block_hash: H256,
229        ) {
230            if previous_states.len() > MAX_PREVIOUS_STATES as usize {
231                if let Some(entry) = previous_states.shift_remove_index(0) {
232                    tracing::debug!("removing archived state for previous block {:#x}", entry.0);
233                }
234            }
235            tracing::debug!("archiving state for {:#x} #{}", block_hash, block_number);
236            previous_states.insert(block_hash, state);
237        }
238
239        let mut storage = self.blockchain.write().await;
240        let new_bytecodes = tx_results
241            .iter()
242            .flat_map(|tr| tr.new_bytecodes.clone())
243            .collect::<Vec<_>>();
244        let aggregation_root = self.read_aggregation_root(&modified_storage_keys);
245        storage.apply_batch(
246            batch_timestamp,
247            base_system_contracts_hashes,
248            tx_results,
249            finished_l1_batch,
250            aggregation_root,
251        );
252
253        // archive current state before we produce new batch/blocks
254        archive_state(
255            &mut self.previous_states,
256            self.fork_storage
257                .inner
258                .read()
259                .unwrap()
260                .raw_storage
261                .state
262                .clone(),
263            storage.current_block,
264            storage.current_block_hash,
265        );
266        storage.apply_block(block, 0);
267
268        // Apply new factory deps
269        for (hash, code) in new_bytecodes {
270            self.fork_storage.store_factory_dep(hash, code)
271        }
272
273        // Apply storage writes
274        for (key, value) in modified_storage_keys {
275            self.fork_storage.set_value(key, value);
276        }
277
278        if let Some(virtual_block) = virtual_block {
279            // archive current state before we produce new batch/blocks
280            archive_state(
281                &mut self.previous_states,
282                self.fork_storage
283                    .inner
284                    .read()
285                    .unwrap()
286                    .raw_storage
287                    .state
288                    .clone(),
289                storage.current_block,
290                storage.current_block_hash,
291            );
292            storage.apply_block(virtual_block, 1);
293        }
294    }

296    fn n_dim_array_key_in_layout(array_key: usize, indices: &[U256]) -> H256 {
297        let mut key: H256 = u256_to_h256(array_key.into());
298
299        for index in indices {
300            key = H256(keccak256(key.as_bytes()));
301            key = u256_to_h256(h256_to_u256(key).overflowing_add(*index).0);
302        }
303
304        key
305    }

307    fn read_aggregation_root(
308        &self,
309        modified_storage_keys: &HashMap<StorageKey, StorageValue>,
310    ) -> H256 {
311        let agg_tree_height_slot = StorageKey::new(
312            AccountTreeId::new(L2_MESSAGE_ROOT_ADDRESS),
313            H256::from_low_u64_be(AGG_TREE_HEIGHT_KEY as u64),
314        );
315
316        let agg_tree_height = modified_storage_keys
317            .get(&agg_tree_height_slot)
318            .copied()
319            .unwrap_or_else(|| {
320                self.fork_storage
321                    .read_value_internal(&agg_tree_height_slot)
322                    .unwrap()
323            });
324        let agg_tree_height = h256_to_u256(agg_tree_height);
325
326        // `nodes[height][0]`
327        let agg_tree_root_hash_key =
328            Self::n_dim_array_key_in_layout(AGG_TREE_NODES_KEY, &[agg_tree_height, U256::zero()]);
329        let agg_tree_root_hash_slot = StorageKey::new(
330            AccountTreeId::new(L2_MESSAGE_ROOT_ADDRESS),
331            agg_tree_root_hash_key,
332        );
333
334        modified_storage_keys
335            .get(&agg_tree_root_hash_slot)
336            .copied()
337            .unwrap_or_else(|| {
338                self.fork_storage
339                    .read_value_internal(&agg_tree_root_hash_slot)
340                    .unwrap()
341            })
342    }

344    pub(super) async fn seal_block(
345        &mut self,
346        tx_batch_execution_result: TxBatchExecutionResult,
347    ) -> AnvilNodeResult<L2BlockNumber> {
348        let TxBatchExecutionResult {
349            tx_results,
350            base_system_contracts_hashes,
351            batch_env,
352            block_ctxs,
353            finished_l1_batch,
354            modified_storage_keys,
355        } = tx_batch_execution_result;
356
357        let mut filters = self.filters.write().await;
358        for tx_result in &tx_results {
359            // TODO: Is this the right place to notify about new pending txs?
360            filters.notify_new_pending_transaction(tx_result.receipt.transaction_hash);
361            for log in &tx_result.receipt.logs {
362                filters.notify_new_log(log, block_ctxs[0].miniblock.into());
363            }
364        }
365        drop(filters);
366
367        let mut transactions = Vec::new();
368        for (index, tx_result) in tx_results.iter().enumerate() {
369            let mut transaction = if let Ok(l2_tx) =
370                <Transaction as TryInto<L2Tx>>::try_into(tx_result.info.tx.clone())
371            {
372                api::Transaction::from(l2_tx)
373            } else {
374                // TODO: Build proper API transaction for upgrade transactions
375                api::Transaction {
376                    hash: tx_result.info.tx.hash(),
377                    ..Default::default()
378                }
379            };
380            transaction.block_hash = Some(block_ctxs[0].hash);
381            transaction.block_number = Some(U64::from(block_ctxs[0].miniblock));
382            transaction.transaction_index = Some(index.into());
383            transaction.l1_batch_number = Some(U64::from(batch_env.number.0));
384            transaction.l1_batch_tx_index = Some(Index::zero());
385            if transaction.transaction_type == Some(U64::zero())
386                || transaction.transaction_type.is_none()
387            {
388                transaction.v = transaction
389                    .v
390                    .map(|v| v + 35 + self.fork_storage.chain_id.as_u64() * 2);
391            }
392            transactions.push(TransactionVariant::Full(transaction));
393        }
394
395        // Build bloom hash
396        let iter = tx_results
397            .iter()
398            .flat_map(|r| r.receipt.logs.iter())
399            .flat_map(|event| {
400                event
401                    .topics
402                    .iter()
403                    .map(|topic| BloomInput::Raw(topic.as_bytes()))
404                    .chain([BloomInput::Raw(event.address.as_bytes())])
405            });
406        let logs_bloom = build_bloom(iter);
407
408        // Calculate how much gas was used across all txs
409        let gas_used = tx_results
410            .iter()
411            .map(|r| r.debug.gas_used)
412            .fold(U256::zero(), |acc, x| acc + x);
413
414        // Construct the block
415        let block = create_block(
416            &batch_env,
417            block_ctxs[0].hash,
418            block_ctxs[0].prev_block_hash,
419            block_ctxs[0].miniblock,
420            block_ctxs[0].timestamp,
421            transactions,
422            gas_used,
423            logs_bloom,
424        );
425
426        // Make sure optional virtual block gets saved too
427        let virtual_block = if block_ctxs.len() == 2 {
428            Some(create_block(
429                &batch_env,
430                block_ctxs[1].hash,
431                block_ctxs[1].prev_block_hash,
432                block_ctxs[1].miniblock,
433                block_ctxs[1].timestamp,
434                vec![],
435                U256::zero(),
436                Bloom::zero(),
437            ))
438        } else {
439            None
440        };
441
442        // Use first block's timestamp as batch timestamp
443        self.apply_batch(
444            batch_env.timestamp,
445            base_system_contracts_hashes,
446            block,
447            virtual_block,
448            tx_results,
449            finished_l1_batch,
450            modified_storage_keys,
451        )
452        .await;
453
454        let mut filters = self.filters.write().await;
455        for block_ctx in &block_ctxs {
456            filters.notify_new_block(block_ctx.hash);
457        }
458        drop(filters);
459
460        Ok(L2BlockNumber(block_ctxs[0].miniblock as u32))
461    }

463    /// Estimates the gas required for a given call request.
464    ///
465    /// # Arguments
466    ///
467    /// * `req` - A `CallRequest` struct representing the call request to estimate gas for.
468    ///
469    /// # Returns
470    ///
471    /// A `Result` with a `Fee` representing the estimated gas related data.
472    pub async fn estimate_gas_impl(&self, req: CallRequest) -> Result<Fee, Web3Error> {
473        let mut request_with_gas_per_pubdata_overridden = req;
474
475        // If not passed, set request nonce to the expected value
476        if request_with_gas_per_pubdata_overridden.nonce.is_none() {
477            let nonce_key = self.storage_key_layout.get_nonce_key(
478                &request_with_gas_per_pubdata_overridden
479                    .from
480                    .unwrap_or_default(),
481            );
482            let full_nonce = self.fork_storage.read_value_alt(&nonce_key).await?;
483            let (account_nonce, _) = decompose_full_nonce(h256_to_u256(full_nonce));
484            request_with_gas_per_pubdata_overridden.nonce = Some(account_nonce);
485        }
486
487        if let Some(ref mut eip712_meta) = request_with_gas_per_pubdata_overridden.eip712_meta {
488            if eip712_meta.gas_per_pubdata == U256::zero() {
489                eip712_meta.gas_per_pubdata =
490                    get_max_gas_per_pubdata_byte(VmVersion::latest()).into();
491            }
492        }
493
494        let is_eip712 = request_with_gas_per_pubdata_overridden
495            .eip712_meta
496            .is_some();
497
498        let mut l2_tx = L2Tx::from_request(
499            request_with_gas_per_pubdata_overridden.into(),
500            MAX_TX_SIZE,
501            self.system_contracts.allow_no_target(),
502        )
503        .map_err(Web3Error::SerializationError)?;
504
505        // Properly format signature
506        if l2_tx.common_data.signature.is_empty() {
507            l2_tx.common_data.signature = vec![0u8; 65];
508            l2_tx.common_data.signature[64] = 27;
509        }
510
511        // The user may not include the proper transaction type during the estimation of
512        // the gas fee. However, it is needed for the bootloader checks to pass properly.
513        if is_eip712 {
514            l2_tx.common_data.transaction_type = TransactionType::EIP712Transaction;
515        }
516
517        l2_tx.common_data.fee.gas_per_pubdata_limit =
518            get_max_gas_per_pubdata_byte(VmVersion::latest()).into();
519
520        self.estimate_gas_inner(l2_tx.into()).await
521    }

523    pub async fn estimate_l1_to_l2_gas_impl(&self, req: CallRequest) -> Result<U256, Web3Error> {
524        let mut request_with_gas_per_pubdata_overridden = req;
525
526        if let Some(ref mut eip712_meta) = request_with_gas_per_pubdata_overridden.eip712_meta {
527            if eip712_meta.gas_per_pubdata == U256::zero() {
528                eip712_meta.gas_per_pubdata =
529                    get_max_gas_per_pubdata_byte(VmVersion::latest()).into();
530            }
531        }
532
533        let l1_tx = L1Tx::from_request(
534            request_with_gas_per_pubdata_overridden,
535            self.system_contracts.allow_no_target(),
536        )
537        .map_err(Web3Error::SerializationError)?;
538
539        Ok(self.estimate_gas_inner(l1_tx.into()).await?.gas_limit)
540    }

542    async fn estimate_gas_inner(&self, mut tx: Transaction) -> Result<Fee, Web3Error> {
543        let fee_input = {
544            let fee_input = self.fee_input_provider.get_batch_fee_input_scaled();
545            // In order for execution to pass smoothly, we need to ensure that block's required gasPerPubdata will be
546            // <= to the one in the transaction itself.
547            adjust_pubdata_price_for_tx(
548                fee_input,
549                tx.gas_per_pubdata_byte_limit(),
550                None,
551                VmVersion::latest(),
552            )
553        };
554
555        let (base_fee, gas_per_pubdata_byte) =
556            derive_base_fee_and_gas_per_pubdata(fee_input, VmVersion::latest());
557        match &mut tx.common_data {
558            ExecuteTransactionCommon::L1(l1_common_data) => {
559                l1_common_data.max_fee_per_gas = base_fee.into();
560            }
561            ExecuteTransactionCommon::L2(l2_common_data) => {
562                l2_common_data.fee.max_fee_per_gas = base_fee.into();
563                l2_common_data.fee.max_priority_fee_per_gas = base_fee.into();
564            }
565            ExecuteTransactionCommon::ProtocolUpgrade(_) => unimplemented!(),
566        }
567
568        let execution_mode = TxExecutionMode::EstimateFee;
569        let (mut batch_env, _) = self.create_l1_batch_env().await;
570        batch_env.fee_input = fee_input;
571
572        let initiator_address = tx.initiator_account();
573        let impersonating = self.impersonation.is_impersonating(&initiator_address);
574        let system_contracts = self
575            .system_contracts
576            .contracts_for_fee_estimate(impersonating)
577            .clone();
578        let system_env = self.create_system_env(system_contracts, execution_mode);
579
580        // When the pubdata cost grows very high, the total gas limit required may become very high as well. If
581        // we do binary search over any possible gas limit naively, we may end up with a very high number of iterations,
582        // which affects performance.
583        //
584        // To optimize for this case, we first calculate the amount of gas needed to cover for the pubdata. After that, we
585        // need to do a smaller binary search that is focused on computational gas limit only.
586        let additional_gas_for_pubdata = if tx.is_l1() {
587            // For L1 transactions the pubdata priced in such a way that the maximal computational
588            // gas limit should be enough to cover for the pubdata as well, so no additional gas is provided there.
589            0u64
590        } else {
591            // For L2 transactions, we estimate the amount of gas needed to cover for the pubdata by creating a transaction with infinite gas limit.
592            // And getting how much pubdata it used.
593
594            // If the transaction has failed with such a large gas limit, we return an API error here right away,
595            // since the inferred gas bounds would be unreliable in this case.
596            let result = self
597                .check_if_executable(
598                    tx.clone(),
599                    gas_per_pubdata_byte,
600                    batch_env.clone(),
601                    system_env.clone(),
602                )
603                .await?;
604
605            if result.statistics.pubdata_published > (MAX_VM_PUBDATA_PER_BATCH as u32) {
606                return Err(Web3Error::SubmitTransactionError(
607                    "exceeds limit for published pubdata".into(),
608                    Default::default(),
609                ));
610            }
611
612            // It is assumed that there is no overflow here
613            (result.statistics.pubdata_published as u64) * gas_per_pubdata_byte
614        };
615
616        // We are using binary search to find the minimal values of gas_limit under which the transaction succeeds
617        let mut lower_bound = 0u64;
618        let mut upper_bound = MAX_L2_TX_GAS_LIMIT;
619        let mut attempt_count = 1;
620
621        tracing::trace!("Starting gas estimation loop");
622        while lower_bound + ESTIMATE_GAS_ACCEPTABLE_OVERESTIMATION < upper_bound {
623            let mid = (lower_bound + upper_bound) / 2;
624            tracing::trace!(
625                "Attempt {} (lower_bound: {}, upper_bound: {}, mid: {})",
626                attempt_count,
627                lower_bound,
628                upper_bound,
629                mid
630            );
631            let try_gas_limit = additional_gas_for_pubdata + mid;
632
633            let estimate_gas_result = self
634                .estimate_gas_step(
635                    tx.clone(),
636                    gas_per_pubdata_byte,
637                    try_gas_limit,
638                    batch_env.clone(),
639                    system_env.clone(),
640                    &self.fork_storage,
641                    &self.system_contracts.boojum,
642                    false,
643                )
644                .tx_result;
645
646            if estimate_gas_result.result.is_failed() {
647                tracing::trace!("Attempt {} FAILED", attempt_count);
648                lower_bound = mid + 1;
649            } else {
650                tracing::trace!("Attempt {} SUCCEEDED", attempt_count);
651                upper_bound = mid;
652            }
653            attempt_count += 1;
654        }
655
656        tracing::trace!("Gas Estimation Values:");
657        tracing::trace!("  Final upper_bound: {}", upper_bound);
658        tracing::trace!(
659            "  ESTIMATE_GAS_SCALE_FACTOR: {}",
660            self.fee_input_provider.estimate_gas_scale_factor
661        );
662        tracing::trace!("  MAX_L2_TX_GAS_LIMIT: {}", MAX_L2_TX_GAS_LIMIT);
663        let tx_body_gas_limit = upper_bound;
664        let suggested_gas_limit = ((upper_bound + additional_gas_for_pubdata) as f32
665            * self.fee_input_provider.estimate_gas_scale_factor)
666            as u64;
667
668        let estimate_gas_result = self
669            .estimate_gas_step(
670                tx.clone(),
671                gas_per_pubdata_byte,
672                suggested_gas_limit,
673                batch_env,
674                system_env,
675                &self.fork_storage,
676                &self.system_contracts.boojum,
677                false,
678            )
679            .tx_result;
680
681        let overhead = derive_overhead(
682            suggested_gas_limit,
683            gas_per_pubdata_byte as u32,
684            tx.encoding_len(),
685            tx.tx_format() as u8,
686            VmVersion::latest(),
687        ) as u64;
688
689        match &estimate_gas_result.result {
690            ExecutionResult::Revert { output } => {
691                let message = output.to_string();
692                let pretty_message = format!(
693                    "execution reverted{}{}",
694                    if message.is_empty() { "" } else { ": " },
695                    message
696                );
697                let data = output.encoded_data();
698
699                let revert_reason: RevertError = output.clone().to_revert_reason().await;
700                let error_report = ExecutionErrorReport::new(&revert_reason, Some(&tx));
701                sh_println!("{}", error_report);
702
703                Err(Web3Error::SubmitTransactionError(pretty_message, data))
704            }
705            ExecutionResult::Halt { reason } => {
706                let message = reason.to_string();
707                let pretty_message = format!(
708                    "execution reverted{}{}",
709                    if message.is_empty() { "" } else { ": " },
710                    message
711                );
712
713                let halt_error: HaltError = reason.clone().to_halt_error().await;
714                let error_report = ExecutionErrorReport::new(&halt_error, Some(&tx));
715                sh_println!("{}", error_report);
716
717                Err(Web3Error::SubmitTransactionError(pretty_message, vec![]))
718            }
719            ExecutionResult::Success { .. } => {
720                let full_gas_limit = match suggested_gas_limit.overflowing_add(overhead) {
721                    (value, false) => value,
722                    (_, true) => {
723                        tracing::info!("Overflow when calculating gas estimation. We've exceeded the block gas limit by summing the following values:");
724                        tracing::info!(
725                            "\tEstimated transaction body gas cost: {}",
726                            tx_body_gas_limit
727                        );
728                        tracing::info!("\tGas for pubdata: {}", additional_gas_for_pubdata);
729                        tracing::info!("\tOverhead: {}", overhead);
730
731                        return Err(Web3Error::SubmitTransactionError(
732                            "exceeds block gas limit".into(),
733                            Default::default(),
734                        ));
735                    }
736                };
737
738                tracing::trace!("Gas Estimation Results");
739                tracing::trace!("  tx_body_gas_limit: {}", tx_body_gas_limit);
740                tracing::trace!(
741                    "  additional_gas_for_pubdata: {}",
742                    additional_gas_for_pubdata
743                );
744                tracing::trace!("  overhead: {}", overhead);
745                tracing::trace!("  full_gas_limit: {}", full_gas_limit);
746                let fee = Fee {
747                    max_fee_per_gas: base_fee.into(),
748                    max_priority_fee_per_gas: 0u32.into(),
749                    gas_limit: full_gas_limit.into(),
750                    gas_per_pubdata_limit: gas_per_pubdata_byte.into(),
751                };
752                Ok(fee)
753            }
754        }
755    }

757    /// Runs fee estimation against a sandbox vm with the given gas_limit.
758    #[allow(clippy::too_many_arguments)]
759    fn estimate_gas_step(
760        &self,
761        mut tx: Transaction,
762        gas_per_pubdata_byte: u64,
763        tx_gas_limit: u64,
764        batch_env: L1BatchEnv,
765        system_env: SystemEnv,
766        fork_storage: &ForkStorage,
767        boojum: &BoojumConfig,
768        trace_calls: bool,
769    ) -> BatchTransactionExecutionResult {
770        // Set gas_limit for transaction
771        let gas_limit_with_overhead = tx_gas_limit
772            + derive_overhead(
773                tx_gas_limit,
774                gas_per_pubdata_byte as u32,
775                tx.encoding_len(),
776                tx.tx_format() as u8,
777                VmVersion::latest(),
778            ) as u64;
779        match &mut tx.common_data {
780            ExecuteTransactionCommon::L1(l1_common_data) => {
781                l1_common_data.gas_limit = gas_limit_with_overhead.into();
782                // Since `tx.execute.value` is supplied by the client and is not checked against the current balance (unlike for L2 transactions),
783                // we may hit an integer overflow. Ditto for protocol upgrade transactions below.
784                let required_funds = (l1_common_data.gas_limit * l1_common_data.max_fee_per_gas)
785                    .checked_add(tx.execute.value)
786                    .unwrap();
787                l1_common_data.to_mint = required_funds;
788            }
789            ExecuteTransactionCommon::L2(l2_common_data) => {
790                l2_common_data.fee.gas_limit = gas_limit_with_overhead.into();
791            }
792            ExecuteTransactionCommon::ProtocolUpgrade(_) => unimplemented!(),
793        }
794
795        let storage = StorageView::new(fork_storage).to_rc_ptr();
796
797        // TODO: core doesn't do this during estimation and fast-fails with validation error instead
798        // We need to explicitly put enough balance into the account of the users
799        let payer = tx.payer();
800        let balance_key = self
801            .storage_key_layout
802            .get_storage_key_for_base_token(&payer);
803        let mut current_balance = h256_to_u256(storage.borrow_mut().read_value(&balance_key));
804        match &mut tx.common_data {
805            ExecuteTransactionCommon::L1(l1_common_data) => {
806                let added_balance = l1_common_data.gas_limit * l1_common_data.max_fee_per_gas;
807                current_balance += added_balance;
808                storage
809                    .borrow_mut()
810                    .set_value(balance_key, u256_to_h256(current_balance));
811            }
812            ExecuteTransactionCommon::L2(l2_common_data) => {
813                let added_balance =
814                    l2_common_data.fee.gas_limit * l2_common_data.fee.max_fee_per_gas;
815                current_balance += added_balance;
816                storage
817                    .borrow_mut()
818                    .set_value(balance_key, u256_to_h256(current_balance));
819            }
820            ExecuteTransactionCommon::ProtocolUpgrade(_) => unimplemented!(),
821        }
822
823        let mut vm = if boojum.use_boojum {
824            let mut vm = BoojumOsVM::<_, HistoryDisabled>::new(
825                batch_env,
826                system_env,
827                storage,
828                // TODO: this might be causing a deadlock.. check..
829                &fork_storage.inner.read().unwrap().raw_storage,
830                boojum,
831            );
832            // Temporary hack - as we update the 'storage' just above, but boojumos loads its full
833            // state from fork_storage (that is not updated).
834            vm.update_inconsistent_keys(&[&balance_key]);
835            AnvilVM::BoojumOs(vm)
836        } else {
837            AnvilVM::ZKSync(Vm::new(batch_env, system_env, storage))
838        };
839
840        delegate_vm!(vm, push_transaction(tx));
841
842        let call_tracer_result = Arc::new(OnceCell::default());
843        let tracer_dispatcher = if trace_calls {
844            let tracers: Vec<Box<dyn MultiVmTracer<StorageView<&ForkStorage>, HistoryDisabled>>> =
845                vec![CallTracer::new(call_tracer_result.clone()).into_tracer_pointer()];
846            TracerDispatcher::from(tracers)
847        } else {
848            Default::default()
849        };
850
851        let tx_result = match &mut vm {
852            AnvilVM::BoojumOs(vm) => {
853                vm.inspect(&mut tracer_dispatcher.into(), InspectExecutionMode::OneTx)
854            }
855            AnvilVM::ZKSync(vm) => {
856                vm.inspect(&mut tracer_dispatcher.into(), InspectExecutionMode::OneTx)
857            }
858        };
859        let call_traces = Arc::try_unwrap(call_tracer_result)
860            .expect("failed extracting call traces")
861            .take()
862            .unwrap_or_default();
863        BatchTransactionExecutionResult {
864            tx_result: Box::new(tx_result),
865            compression_result: Ok(()),
866            call_traces,
867        }
868    }

870    async fn check_if_executable(
871        &self,
872        tx: Transaction,
873        gas_per_pubdata_byte: u64,
874        batch_env: L1BatchEnv,
875        system_env: SystemEnv,
876    ) -> Result<VmExecutionResultAndLogs, Web3Error> {
877        let verbosity = get_shell().verbosity;
878        let mut known_addresses = known_addresses_after_transaction(&tx);
879        let BatchTransactionExecutionResult {
880            tx_result,
881            call_traces,
882            ..
883        } = self.estimate_gas_step(
884            tx.clone(),
885            gas_per_pubdata_byte,
886            // TODO: check what the max value should be here.
887            MAX_L2_TX_GAS_LIMIT,
888            batch_env,
889            system_env,
890            &self.fork_storage,
891            &self.system_contracts.boojum,
892            true,
893        );
894
895        let error = match tx_result.result {
896            ExecutionResult::Success { .. } => {
897                // Transaction is executable with max gas, proceed with gas estimation
898                return Ok(*tx_result);
899            }
900            ExecutionResult::Revert { ref output } => {
901                let message = output.to_string();
902                let pretty_message = format!(
903                    "execution reverted{}{}",
904                    if message.is_empty() { "" } else { ": " },
905                    message
906                );
907                let data = output.encoded_data();
908
909                if verbosity >= 1 {
910                    let revert_reason: RevertError = output.clone().to_revert_reason().await;
911                    let error_report = ExecutionErrorReport::new(&revert_reason, Some(&tx));
912                    sh_println!(
913                        "{}: {}\n{}",
914                        "error".red().bold(),
915                        "Gas estimation encountered unexecutable transaction".red(),
916                        error_report
917                    );
918                }
919
920                Web3Error::SubmitTransactionError(pretty_message, data)
921            }
922            ExecutionResult::Halt { ref reason } => {
923                let message = reason.to_string();
924                let pretty_message = format!(
925                    "execution reverted{}{}",
926                    if message.is_empty() { "" } else { ": " },
927                    message
928                );
929
930                if verbosity >= 1 {
931                    let halt_error: HaltError = reason.clone().to_halt_error().await;
932                    let error_report = ExecutionErrorReport::new(&halt_error, Some(&tx));
933                    sh_println!(
934                        "{}: {}\n{}",
935                        "error".red().bold(),
936                        "Gas estimation encountered unexecutable transaction".red(),
937                        error_report
938                    );
939                }
940
941                Web3Error::SubmitTransactionError(pretty_message, vec![])
942            }
943        };
944
945        if !call_traces.is_empty() && verbosity >= 2 {
946            let mut builder = CallTraceDecoderBuilder::default();
947
948            builder = builder.with_signature_identifier(
949                SignaturesIdentifier::new(
950                    Some(self.config.get_cache_dir().into()),
951                    self.config.offline,
952                )
953                .map_err(|err| anyhow::anyhow!("Failed to create SignaturesIdentifier: {err:#}"))?,
954            );
955
956            let decoder = builder.build();
957            let mut arena = build_call_trace_arena(&call_traces, &tx_result);
958            decode_trace_arena(&mut arena, &decoder).await;
959
960            extract_addresses(&arena, &mut known_addresses);
961
962            let filtered_arena = filter_call_trace_arena(&arena, verbosity);
963            let trace_output = render_trace_arena_inner(&filtered_arena, false);
964            sh_println!("\nTraces:\n{}", trace_output);
965        }
966        Err(error)
967    }

1052    pub async fn load_state(&mut self, state: VersionedState) -> StateLoaderResult<bool> {
1053        let mut storage = self.blockchain.write().await;
1054        if storage.blocks.len() > 1 {
1055            tracing::debug!(
1056                blocks = storage.blocks.len(),
1057                "node has existing state; refusing to load new state"
1058            );
1059            return Err(StateLoaderError::LoadingStateOverExistingState);
1060        }
1061        let state = match state {
1062            VersionedState::V1 { state, .. } => state,
1063            VersionedState::Unknown { version } => {
1064                return Err(StateLoaderError::UnknownStateVersion {
1065                    version: version.into(),
1066                })
1067            }
1068        };
1069        if state.blocks.is_empty() {
1070            tracing::debug!("new state has no blocks; refusing to load");
1071            return Err(StateLoaderError::LoadEmptyState);
1072        }
1073
1074        storage.load_blocks(&mut self.time, state.blocks);
1075        storage.load_transactions(state.transactions);
1076        self.fork_storage.load_state(state.fork_storage);
1077
1078        tracing::trace!(
1079            states = state.historical_states.len(),
1080            "loading historical states from supplied state"
1081        );
1082        self.previous_states.extend(
1083            state
1084                .historical_states
1085                .into_iter()
1086                .map(|(k, v)| (k, v.0.into_iter().collect())),
1087        );
1088
1089        Ok(true)
1090    }

1092    pub async fn get_storage_at_block(
1093        &self,
1094        address: Address,
1095        idx: U256,
1096        block: Option<api::BlockIdVariant>,
1097    ) -> Result<H256, Web3Error> {
1098        let storage_key = StorageKey::new(AccountTreeId::new(address), u256_to_h256(idx));
1099        let storage = self.blockchain.read().await;
1100
1101        let block_number = block
1102            .map(|block| match block {
1103                BlockIdVariant::BlockNumber(block_number) => Ok(utils::to_real_block_number(
1104                    block_number,
1105                    U64::from(storage.current_block.0),
1106                )),
1107                BlockIdVariant::BlockNumberObject(o) => Ok(utils::to_real_block_number(
1108                    o.block_number,
1109                    U64::from(storage.current_block.0),
1110                )),
1111                BlockIdVariant::BlockHashObject(o) => storage
1112                    .blocks
1113                    .get(&o.block_hash)
1114                    .map(|block| block.number)
1115                    .ok_or_else(|| {
1116                        tracing::error!("unable to map block number to hash #{:#x}", o.block_hash);
1117                        Web3Error::InternalError(anyhow::Error::msg(
1118                            "Failed to map block number to hash.",
1119                        ))
1120                    }),
1121            })
1122            .unwrap_or_else(|| Ok(U64::from(storage.current_block.0)))?;
1123        // FIXME: Conversion mess above
1124        let block_number = L2BlockNumber(block_number.as_u32());
1125
1126        if block_number == storage.current_block {
1127            match self.fork_storage.read_value_internal(&storage_key) {
1128                Ok(value) => Ok(H256(value.0)),
1129                Err(error) => Err(Web3Error::InternalError(anyhow::anyhow!(
1130                    "failed to read storage: {}",
1131                    error
1132                ))),
1133            }
1134        } else if let Some(block_hash) = storage.hashes.get(&block_number) {
1135            let state = self
1136                .previous_states
1137                .get(block_hash)
1138                .ok_or_else(|| Web3Error::PrunedBlock(block_number))?;
1139            if let Some(value) = state.get(&storage_key) {
1140                return Ok(*value);
1141            }
1142            // Block was produced locally but slot hasn't been touched since we forked
1143            Ok(self.fork.get_storage_at_forked(address, idx).await?)
1144        } else {
1145            // Block was not produced locally so we assume it comes from fork
1146            Ok(self.fork.get_storage_at(address, idx, block).await?)
1147        }
1148    }

1150    pub async fn reset(&mut self, fork_client_opt: Option<ForkClient>) {
1151        let fork_details = fork_client_opt.as_ref().map(|client| &client.details);
1152        let blockchain = Blockchain::new(
1153            self.blockchain.protocol_version,
1154            fork_details,
1155            self.config.genesis.as_ref(),
1156            self.config.genesis_timestamp,
1157        );
1158        let blockchain_storage = blockchain.read().await.clone();
1159        drop(std::mem::replace(
1160            &mut *self.blockchain.write().await,
1161            blockchain_storage,
1162        ));
1163
1164        self.time.set_current_timestamp_unchecked(
1165            fork_details
1166                .map(|fd| fd.block_timestamp)
1167                .unwrap_or(NON_FORK_FIRST_BLOCK_TIMESTAMP),
1168        );
1169
1170        drop(std::mem::take(&mut *self.filters.write().await));
1171
1172        self.fork.reset_fork_client(fork_client_opt);
1173        let fork_storage = ForkStorage::new(
1174            self.fork.clone(),
1175            self.config.system_contracts_options,
1176            self.blockchain.protocol_version,
1177            self.config.chain_id,
1178            self.config.system_contracts_path.as_deref(),
1179        );
1180        let mut old_storage = self.fork_storage.inner.write().unwrap();
1181        let mut new_storage = fork_storage.inner.write().unwrap();
1182        old_storage.raw_storage = std::mem::take(&mut new_storage.raw_storage);
1183        old_storage.value_read_cache = std::mem::take(&mut new_storage.value_read_cache);
1184        old_storage.factory_dep_cache = std::mem::take(&mut new_storage.factory_dep_cache);
1185        self.fork_storage.chain_id = fork_storage.chain_id;
1186        drop(old_storage);
1187        drop(new_storage);
1188
1189        self.rich_accounts.clear();
1190        self.previous_states.clear();
1191
1192        let rich_addresses = itertools::chain!(
1193            self.config
1194                .genesis_accounts
1195                .iter()
1196                .map(|acc| H160::from_slice(acc.address().as_ref())),
1197            self.config
1198                .signer_accounts
1199                .iter()
1200                .map(|acc| H160::from_slice(acc.address().as_ref())),
1201            LEGACY_RICH_WALLETS
1202                .iter()
1203                .map(|(address, _)| H160::from_str(address).unwrap()),
1204            RICH_WALLETS
1205                .iter()
1206                .map(|(address, _, _)| H160::from_str(address).unwrap()),
1207        )
1208        .collect::<Vec<_>>();
1209        for address in rich_addresses {
1210            self.set_rich_account(address, self.config.genesis_balance);
1211        }
1212    }

1214    /// Adds a lot of tokens to a given account with a specified balance.
1215    pub fn set_rich_account(&mut self, address: H160, balance: U256) {
1216        let key = self
1217            .storage_key_layout
1218            .get_storage_key_for_base_token(&address);
1219
1220        let keys = {
1221            let mut storage_view = StorageView::new(&self.fork_storage);
1222            // Set balance to the specified amount
1223            storage_view.set_value(key, u256_to_h256(balance));
1224            storage_view.modified_storage_keys().clone()
1225        };
1226
1227        for (key, value) in keys.iter() {
1228            self.fork_storage.set_value(*key, *value);
1229        }
1230        self.rich_accounts.insert(address);
1231    }

1243/// Keeps track of a block's batch number, miniblock number and timestamp.
1244/// Useful for keeping track of the current context when creating multiple blocks.
1245#[derive(Debug, Clone, Default)]
1246pub struct BlockContext {
1247    pub hash: H256,
1248    pub batch: u32,
1249    pub miniblock: u64,
1250    pub timestamp: u64,
1251    pub prev_block_hash: H256,
1252}
1253
1254impl BlockContext {
1255    /// Create the next batch instance that uses the same batch number, and has all other parameters incremented by `1`.
1256    pub(super) fn new_block(&self, time: &mut Time) -> BlockContext {
1257        Self {
1258            hash: H256::zero(),
1259            batch: self.batch,
1260            miniblock: self.miniblock.saturating_add(1),
1261            timestamp: time.advance_timestamp(),
1262            prev_block_hash: self.hash,
1263        }
1264    }
1265}
