101/// Current commitment encoding version by protocol version.
102pub fn supported_encoding_version(batch: &L1BatchWithMetadata) -> u8 {
103    if batch
104        .header
105        .protocol_version
106        .unwrap_or_default()
107        .is_pre_interop_fast_blocks()
108    {
109        0
110    } else {
111        1
112    }
113}

115/// Builds a Solidity function call to `commitBatchesSharedBridge` as expected by `IExecutor.sol`.

141    } else {

152/// `commitBatchesSharedBridge` expects the rest of calldata to be of very specific form. This
153/// function makes sure last committed batch and new batch are encoded correctly (assumes post gateway).
154fn commit_calldata(
155    last_committed_l1_batch: &L1BatchWithMetadata,
156    batch: &L1BatchWithMetadata,
157) -> Vec<u8> {
158    let commit_batch_info = IExecutor::CommitBatchInfo::from((
159        batch.header.number.0 as u64,
160        batch.header.timestamp,
161        batch.metadata.rollup_last_leaf_index,
162        alloy::primitives::FixedBytes::<32>::from(batch.metadata.root_hash.0),
163        alloy::primitives::U256::from(batch.header.l1_tx_count),
164        alloy::primitives::FixedBytes::<32>::from(batch.header.priority_ops_onchain_data_hash().0),
165        alloy::primitives::FixedBytes::<32>::from(
166            batch
167                .metadata
168                .bootloader_initial_content_commitment
169                .unwrap()
170                .0,
171        ),
172        alloy::primitives::FixedBytes::<32>::from(
173            batch.metadata.events_queue_commitment.unwrap().0,
174        ),
175        alloy::primitives::Bytes::from(serialize_commitments(&batch.header.system_logs)),
176        // Our DA input consists only of state diff hash. Executor is patched to not use anything else.
177        alloy::primitives::Bytes::from(
178            batch
179                .metadata
180                .state_diff_hash
181                .expect("Failed to get state_diff_hash from metadata")
182                .0,
183        ),
184    ));
185
186    let encoded_data = if batch
187        .header
188        .protocol_version
189        .unwrap_or_default()
190        .is_pre_interop_fast_blocks()
191    {
192        let stored_batch_info = IExecutor::LegacyStoredBatchInfo::from(last_committed_l1_batch);
193        let last_batch_hash = H256(keccak256(stored_batch_info.abi_encode_params().as_slice()));
194        tracing::info!(?last_batch_hash, "preparing commit calldata");
195        (stored_batch_info, vec![commit_batch_info]).abi_encode_params()
196    } else {
197        let stored_batch_info = IExecutor::StoredBatchInfo::from(last_committed_l1_batch);
198        let last_batch_hash = H256(keccak256(stored_batch_info.abi_encode_params().as_slice()));
199        tracing::info!(?last_batch_hash, "preparing commit calldata");
200        (stored_batch_info, vec![commit_batch_info]).abi_encode_params()
201    };
202
203    // Prefixed by current encoding version as expected by protocol
204    [[supported_encoding_version(batch)].to_vec(), encoded_data]
205        .concat()
206        .to_vec()
207}

302/// `executeBatchesSharedBridge` expects the rest of calldata to be of very specific form. This
303/// function makes sure batch and its priority operations are encoded correctly (assumes post gateway).
304fn execute_calldata(
305    batch: &L1BatchWithMetadata,
306    l1_tx_merkle_tree: &MiniMerkleTree<L1Tx>,
307) -> Vec<u8> {
308    let count = batch.header.l1_tx_count as usize;
309    let priority_ops_proofs = if count > 0 {
310        let (_, left, right) = l1_tx_merkle_tree.merkle_root_and_paths_for_range(..count);
311        let hashes = l1_tx_merkle_tree.hashes_prefix(count);
312        vec![PriorityOpsBatchInfo {
313            leftPath: left
314                .into_iter()
315                .map(Option::unwrap_or_default)
316                .map(|hash| TxHash::from(hash.0))
317                .collect(),
318            rightPath: right
319                .into_iter()
320                .map(Option::unwrap_or_default)
321                .map(|hash| TxHash::from(hash.0))
322                .collect(),
323            itemHashes: hashes
324                .into_iter()
325                .map(|hash| TxHash::from(hash.0))
326                .collect(),
327        }]
328    } else {
329        vec![PriorityOpsBatchInfo {
330            leftPath: vec![],
331            rightPath: vec![],
332            itemHashes: vec![],
333        }]
334    };
335
336    let encoded_data = if batch
337        .header
338        .protocol_version
339        .unwrap_or_default()
340        .is_pre_interop_fast_blocks()
341    {
342        let batches_arg = vec![IExecutor::LegacyStoredBatchInfo::from(batch)];
343        (batches_arg, priority_ops_proofs).abi_encode_params()
344    } else {
345        let dependency_roots: Vec<Vec<InteropRoot>> = vec![vec![]];
346        let batches_arg = vec![IExecutor::StoredBatchInfo::from(batch)];
347        (batches_arg, priority_ops_proofs, dependency_roots).abi_encode_params()
348    };
349
350    // Prefixed by current encoding version as expected by protocol
351    [[supported_encoding_version(batch)].to_vec(), encoded_data]
352        .concat()
353        .to_vec()
